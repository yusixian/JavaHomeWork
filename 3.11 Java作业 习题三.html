（2）编写一个应用程序求100以内的全部素数

public class Main {
	public static boolean isPrime(int x) {
		if(x == 1) return false;	//1不是素数
		int k = (int) Math.sqrt(x);
		for(int i = 2; i <= k; ++i) {
			if(x % i == 0) return false;
		}
		return true;
	}
	public static void main(String[] args) {
		for(int i = 1; i <= 100; ++i) {
			if(isPrime(i)) 
				System.out.println(i);
		}
	}
}

（3）分别用do-while和for循环计算1+1/2!+1/3!+1/4!+…的前20项和

public class Main {
	public static void main(String[] args) {
		double [] f = new double[25];	//f[i]表示i的阶乘
		f[0] = f[1] = 1;
		for(int i = 2; i <= 20; ++i) {
			f[i] = f[i-1] * i;
		}
		
		double sum1 = 0, sum2 = 0;
		
		int i = 1; 
		do {
			sum1 += 1/f[i];
			++i;
		} while(i < 21);
		
		for(int j = 1; j <= 20; ++j) {
			sum2 += 1/f[j];
		}
		System.out.println("do-while计算得：" + sum1);
		System.out.println("for循环计算得：" + sum2);
	}
}

（6）编写应用程序，输出满足1+2+3+…+n<8888的最大正整数n

public class Main {
	public static void main(String[] args) {
		int n = 1;
		int sum = 0;
		for(;; ++n){
			sum += n;
			if(sum >= 8888) break;
		}
		System.out.println(n-1);
	}
}
